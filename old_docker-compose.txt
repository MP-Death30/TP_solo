## docker-compose.yml
version: '3.8'
services:
  app:
    build: .
    ports:
      - "8000:8000"
    volumes:
      - .:/app
    depends_on:
      - mariadb
  mariadb:
    image: mariadb:10.5
    environment:
      MYSQL_USER: root_user
      MYSQL_ROOT_PASSWORD: root
      MYSQL_DATABASE: events_db
    ports:
      - "3306:3306"
  adminer:
    image: adminer
    ports:
      - "8080:8080"
